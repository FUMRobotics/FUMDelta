DEF movement()
DECL POS P1
DECL POS P2
DECL INT I, J, K
DECL BOOLEAN SW
I = 1
SW = false
I = I - 1
P1 = {X 1025,Y 0,Z 1480,A 0,B 0,C 10}
P2 = {X -1025,Y 0,Z -1480,A 0,B 0,C 10}
PTP {POS:X 1025,Y 0,Z 1480,A 0,B 0,C 0}
LIN {X 1030,Y 350,Z 1300,A 160,B 45,C 130}
FOR I = 1 TO 5
FOR J = 1 TO 10
PTP{AXIS: A1 0,A2 -90,A3 90,A4 0}
ENDFOR
IF SW <> true THEN
CIRC P1,P2
ELSE
I = 10
ENDIF
ENDFOR
END
